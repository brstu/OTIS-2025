cmake_minimum_required(VERSION 3.15)
project(task_03_ii02822) 

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Основная программа
add_executable(lab3_ii02822
    main.cpp
    pid_controller.cpp
    model.cpp
)

# GoogleTest через FetchContent
include(FetchContent)
FetchContent_Declare(
  googletest
  # Используйте ZIP архив вместо GIT репозитория
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
  DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)
FetchContent_MakeAvailable(googletest)

enable_testing()

# Тесты
add_executable(testlab3_runner
    ../test/pid_controller_test.cpp
    pid_controller.cpp
    model.cpp
)

target_link_libraries(testlab3_runner gtest_main pthread)
include(GoogleTest)
gtest_discover_tests(testlab3_runner)

message(STATUS "Конфигурация проекта task_03_ii02822 завершена")
