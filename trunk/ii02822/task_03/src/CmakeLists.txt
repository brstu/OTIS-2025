cmake_minimum_required(VERSION 3.15)
project(task_03_ii02822) 

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Основная программа
add_executable(lab3_ii02822
    main.cpp
    pid_controller.cpp
    model.cpp
)

# GoogleTest через FetchContent
include(FetchContent)
FetchContent_Declare(
  googletest
  GIT_REPOSITORY https://github.com/google/googletest.git
  GIT_TAG        v1.14.0
)
FetchContent_MakeAvailable(googletest)

enable_testing()

# ИСПРАВЛЕННЫЙ путь к тестам - используем абсолютный путь
add_executable(testlab3_runner
    ${CMAKE_CURRENT_SOURCE_DIR}/../test/pid_controller_test.cpp  # ← ИСПРАВЛЕНО
    pid_controller.cpp
    model.cpp
)

target_link_libraries(testlab3_runner gtest_main pthread)
include(GoogleTest)
gtest_discover_tests(testlab3_runner)

message(STATUS "Конфигурация проекта task_03_ii02822 завершена")
