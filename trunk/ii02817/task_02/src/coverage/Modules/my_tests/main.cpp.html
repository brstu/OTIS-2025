        std::vector<double> ans(steps);

            ans[0] = t_curr;
        }
        for (int i = 1; i < steps; i++)

        }
        return ans;
    }

    {
        if (steps <= 0)

        std::vector<double> ans(steps);

        if (steps >= 2)
        {
            ans[1] = t_curr;

        {

        }
        return ans;
    }
